写法 A：find()

int Grid::getLayerId(const std::string& layerName) const {
    auto it = layerIDwithName.find(layerName);
    if (it != layerIDwithName.end()) {
        return it->second;              // ✅ 用迭代器取值，不用 []
    } else {
        std::cerr << "Error: Layer not found: " << layerName << "\n";
        return -1;
    }
}
